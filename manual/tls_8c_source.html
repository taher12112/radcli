78 static ssize_t tls_sendto(void *ptr, int sockfd,
79  const void *buf, size_t len,
80  int flags, const struct sockaddr *dest_addr,
81  socklen_t addrlen)
82 {
83  tls_st *st = ptr;
84  int ret;
85 
86  if (st->ctx.need_restart != 0) {
87  restart_session(st->rh, st);
88  }
89 
90  ret = gnutls_record_send(st->ctx.session, buf, len);
91  if (ret == GNUTLS_E_AGAIN || ret == GNUTLS_E_INTERRUPTED) {
92  errno = EINTR;
93  return -1;
94  }
95 
96  if (ret < 0) {
97  rc_log(LOG_ERR, "%s: error in sending: %s", __func__,
98  gnutls_strerror(ret));
99  errno = EIO;
100  st->ctx.need_restart = 1;
101  return -1;
102  }
103 
104  st->ctx.last_msg = time(0);
105  return ret;
106 }

122 static ssize_t tls_recvfrom(void *ptr, int sockfd,
123  void *buf, size_t len,
124  int flags, struct sockaddr *src_addr,
125  socklen_t * addrlen)
126 {
127  tls_st *st = ptr;
128  int ret;
129 
130  ret = gnutls_record_recv(st->ctx.session, buf, len);
131  if (ret == GNUTLS_E_AGAIN || ret == GNUTLS_E_INTERRUPTED ||
132  ret == GNUTLS_E_HEARTBEAT_PING_RECEIVED || ret == GNUTLS_E_HEARTBEAT_PONG_RECEIVED) {
133  errno = EINTR;
134  return -1;
135  }
136 
137  if (ret == GNUTLS_E_WARNING_ALERT_RECEIVED) {
138  rc_log(LOG_ERR, "%s: received: %s", __func__,
139  gnutls_strerror(ret), gnutls_alert_get_name(gnutls_alert_get(st->ctx.session)));
140  errno = EINTR;
141  return -1;
142  }
143 
144  /* RFC6614 says: "After the TLS session is established, RADIUS packet payloads are
145  * exchanged over the encrypted TLS tunnel. In RADIUS/UDP, the
146  * packet size can be determined by evaluating the size of the
147  * datagram that arrived. Due to the stream nature of TCP and TLS,
148  * this does not hold true for RADIUS/TLS packet exchange.",
149  *
150  * That is correct in principle but it fails to associate the length with
151  * the TLS record boundaries. Here, when in TLS, we assume that a single TLS
152  * record holds a single radius packet. It wouldn't make sense anyway to send
153  * multiple TLS records for a single packet.
154  */
155 
156  if (ret <= 0) {
157  rc_log(LOG_ERR, "%s: error in receiving: %s", __func__,
158  gnutls_strerror(ret));
159  errno = EIO;
160  st->ctx.need_restart = 1;
161  return -1;
162  }
163 
164  st->ctx.last_msg = time(0);
165  return ret;
166 }

171 static int cert_verify_callback(gnutls_session_t session)
172 {
173  unsigned int status;
174  int ret;
175  struct tls_int_st *ctx;
176  const char *hostname;
177  gnutls_datum_t out;
178 
179  /* read hostname */
180  ctx = gnutls_session_get_ptr(session);
181  if (ctx == NULL)
182  return GNUTLS_E_CERTIFICATE_ERROR;
183 
184  if (ctx->skip_hostname_check)
185  ret = gnutls_certificate_verify_peers2(session, &status);
186  else
187  ret = gnutls_certificate_verify_peers3(session, ctx->hostname, &status);
188  if (ret < 0) {
189  rc_log(LOG_ERR, "%s: error in certificate verification: %s",
190  __func__, gnutls_strerror(ret));
191  return GNUTLS_E_CERTIFICATE_ERROR;
192  }
193 
194  if (status != 0) {
195  ret =
196  gnutls_certificate_verification_status_print(status,
197  gnutls_certificate_type_get
198  (session),
199  &out, 0);
200  if (ret < 0) {
201  return GNUTLS_E_CERTIFICATE_ERROR;
202  }
203  rc_log(LOG_INFO, "%s: certificate: %s", __func__, out.data);
204  gnutls_free(out.data);
205  return GNUTLS_E_CERTIFICATE_ERROR;
206  }
207 
208  return 0;
209 }

223 static int init_session(rc_handle *rh, tls_int_st *ses,
224  const char *hostname, unsigned port,
225  struct sockaddr_storage *our_sockaddr,
226  int timeout,
227  unsigned secflags)
228 {
229  int sockfd, ret, e;
230  const char *pskkey = NULL;
231  struct addrinfo *info;
232  char *p;
233  unsigned flags = 0;
234  unsigned cred_set = 0;
235  tls_st *st = rh->so.ptr;
236 
237  ses->sockfd = -1;
238  ses->init = 1;
239 
240  pthread_mutex_init(&ses->lock, NULL);
241  sockfd = socket(our_sockaddr->ss_family, (secflags&SEC_FLAG_DTLS)?SOCK_DGRAM:SOCK_STREAM, 0);
242  if (sockfd < 0) {
243  rc_log(LOG_ERR,
244  "%s: cannot open socket", __func__);
245  ret = -1;
246  goto cleanup;
247  }
248 
249  if (our_sockaddr->ss_family == AF_INET)
250  ((struct sockaddr_in *)our_sockaddr)->sin_port = 0;
251  else
252  ((struct sockaddr_in6 *)our_sockaddr)->sin6_port = 0;
253 
254  ses->sockfd = sockfd;
255 
256  /* Initialize DTLS */
257 
258  flags = GNUTLS_CLIENT;
259  if (secflags&SEC_FLAG_DTLS)
260  flags |= GNUTLS_DATAGRAM;
261  ret = gnutls_init(&ses->session, flags);
262  if (ret < 0) {
263  rc_log(LOG_ERR,
264  "%s: error in gnutls_init(): %s", __func__, gnutls_strerror(ret));
265  ret = -1;
266  goto cleanup;
267  }
268 
269  memcpy(&ses->our_sockaddr, our_sockaddr, sizeof(*our_sockaddr));
270  if (!(secflags&SEC_FLAG_DTLS)) {
271  if (timeout > 0) {
272  gnutls_handshake_set_timeout(ses->session, timeout*1000);
273  } else {
274  gnutls_handshake_set_timeout(ses->session, GNUTLS_DEFAULT_HANDSHAKE_TIMEOUT);
275  }
276  } else { /* DTLS */
277  if (timeout > 0)
278  gnutls_dtls_set_timeouts(ses->session, 1000, timeout*1000);
279  }
280 
281  gnutls_transport_set_int(ses->session, sockfd);
282  gnutls_session_set_ptr(ses->session, ses);
283  /* we only initiate heartbeat messages */
284  gnutls_heartbeat_enable(ses->session, GNUTLS_HB_LOCAL_ALLOWED_TO_SEND);
285 
286  p = rc_conf_str(rh, "tls-verify-hostname");
287  if (p && (strcasecmp(p, "false") == 0 || strcasecmp(p, "no"))) {
288  ses->skip_hostname_check = 1;
289  }
290 
291  if (st && st->psk_cred) {
292  cred_set = 1;
293  gnutls_credentials_set(ses->session,
294  GNUTLS_CRD_PSK, st->psk_cred);
295 
296  ret = gnutls_priority_set_direct(ses->session, "NORMAL:-KX-ALL:+ECDHE-PSK:+DHE-PSK:+PSK:-VERS-TLS1.0", NULL);
297  if (ret < 0) {
298  ret = -1;
299  rc_log(LOG_ERR,
300  "%s: error in setting PSK priorities: %s",
301  __func__, gnutls_strerror(ret));
302  goto cleanup;
303 
304  cred_set = 1;
305  }
306  } else if (st) {
307  cred_set = 1;
308  if (st->x509_cred) {
309  gnutls_credentials_set(ses->session,
310  GNUTLS_CRD_CERTIFICATE,
311  st->x509_cred);
312  }
313 
314  gnutls_set_default_priority(ses->session);
315  }
316 
317  gnutls_server_name_set(ses->session, GNUTLS_NAME_DNS,
318  hostname, strlen(hostname));
319 
320  info =
321  rc_getaddrinfo(hostname, PW_AI_AUTH);
322  if (info == NULL) {
323  ret = -1;
324  rc_log(LOG_ERR, "%s: cannot resolve %s", __func__,
325  hostname);
326  goto cleanup;
327  }
328 
329  if (port != 0) {
330  if (info->ai_addr->sa_family == AF_INET)
331  ((struct sockaddr_in *)info->ai_addr)->sin_port =
332  htons(port);
333  else
334  ((struct sockaddr_in6 *)info->ai_addr)->sin6_port =
335  htons(port);
336  } else {
337  rc_log(LOG_ERR, "%s: no port specified for server %s",
338  __func__, hostname);
339  ret = -1;
340  goto cleanup;
341  }
342 
343  strlcpy(ses->hostname, hostname, sizeof(ses->hostname));
344  ses->port = port;
345 
346  if (cred_set == 0) {
347  rc_log(LOG_CRIT,
348  "%s: neither tls-ca-file or a PSK key are configured",
349  __func__);
350  ret = -1;
351  goto cleanup;
352  }
353 
354  /* we connect since we are talking to a single server */
355  ret = connect(sockfd, info->ai_addr, info->ai_addrlen);
356  freeaddrinfo(info);
357  if (ret == -1) {
358  e = errno;
359  ret = -1;
360  rc_log(LOG_CRIT, "%s: cannot connect to %s: %s",
361  __func__, hostname, strerror(e));
362  goto cleanup;
363  }
364 
365  rc_log(LOG_DEBUG,
366  "%s: performing TLS/DTLS handshake with [%s]:%d",
367  __func__, hostname, port);
368  do {
369  ret = gnutls_handshake(ses->session);
370  if (ret == GNUTLS_E_LARGE_PACKET)
371  break;
372  } while (ret < 0 && gnutls_error_is_fatal(ret) == 0);
373 
374  if (ret < 0) {
375  rc_log(LOG_ERR, "%s: error in handshake: %s",
376  __func__, gnutls_strerror(ret));
377  ret = -1;
378  goto cleanup;
379  }
380 
381  return 0;
382  cleanup:
383  deinit_session(ses);
384  return ret;
385 
386 }

392 static void restart_session(rc_handle *rh, tls_st *st)
393 {
394  struct tls_int_st tmps;
395  time_t now = time(0);
396  int ret;
397  int timeout;
398 
399  if (now - st->ctx.last_restart < TIME_ALIVE)
400  return;
401 
402  st->ctx.last_restart = now;
403 
404  timeout = rc_conf_int(rh, "radius_timeout");
405 
406  /* reinitialize this session */
407  ret = init_session(rh, &tmps, st->ctx.hostname, st->ctx.port, &st->ctx.our_sockaddr, timeout, st->flags);
408  if (ret < 0) {
409  rc_log(LOG_ERR, "%s: error in re-initializing DTLS");
410  return;
411  }
412 
413  if (tmps.sockfd == st->ctx.sockfd)
414  st->ctx.sockfd = -1;
415  deinit_session(&st->ctx);
416  memcpy(&st->ctx, &tmps, sizeof(tmps));
417  st->ctx.need_restart = 0;
418 
419  return;
420 }

454 int rc_check_tls(rc_handle * rh)
455 {
456  tls_st *st;
457  time_t now = time(0);
458  int ret;
459  unsigned type;
460 
461  if (rh->so_set != SOCKETS_TLS && rh->so_set != SOCKETS_DTLS)
462  return 0;
463 
464  st = rh->so.ptr;
465 
466  if (st->ctx.init != 0) {
467  if (st->ctx.need_restart != 0) {
468  restart_session(rh, st);
469  } else if (now - st->ctx.last_msg > TIME_ALIVE) {
470  ret = gnutls_heartbeat_ping(st->ctx.session, 64, 4, GNUTLS_HEARTBEAT_WAIT);
471  if (ret < 0) {
472  restart_session(rh, st);
473  }
474  st->ctx.last_msg = now;
475  }
476  }
477  return 0;
478 }

507 {

510  struct sockaddr_storage our_sockaddr;
511  unsigned i;
512  const char *ca_file = rc_conf_str(rh, "tls-ca-file");
513  const char *cert_file = rc_conf_str(rh, "tls-cert-file");
514  const char *key_file = rc_conf_str(rh, "tls-key-file");
515  const char *pskkey = NULL;
516  SERVER *authservers;
517  char hostname[256]; /* server's hostname */
518  unsigned port; /* server's port */
519 
520  memset(&rh->so, 0, sizeof(rh->so));
521 
522  if (flags & SEC_FLAG_DTLS) {
523  rh->so_set = SOCKETS_DTLS;
524  rh->so.static_secret = DEFAULT_DTLS_SECRET;
525  } else {
526  rh->so_set = SOCKETS_TLS;
527  rh->so.static_secret = DEFAULT_TLS_SECRET;
528  }
529 
530  rc_own_bind_addr(rh, &our_sockaddr);
531 
532  st = calloc(1, sizeof(tls_st));
533  if (st == NULL) {
534  ret = -1;
535  goto cleanup;
536  }
537

539  st->flags = flags;
540 
541  rh->so.ptr = st;
542

544  ret = gnutls_certificate_allocate_credentials(&st->x509_cred);
545  if (ret < 0) {
546  ret = -1;
547  rc_log(LOG_ERR,
548  "%s: error in setting X.509 credentials: %s",
549  __func__, gnutls_strerror(ret));
550  goto cleanup;
551  }
552 
553  if (ca_file) {
554  ret =
555  gnutls_certificate_set_x509_trust_file(st->x509_cred,
556  ca_file,
557  GNUTLS_X509_FMT_PEM);
558  if (ret < 0) {
559  ret = -1;
560  rc_log(LOG_ERR,
561  "%s: error in setting X.509 trust file: %s",
562  __func__, gnutls_strerror(ret));
563  goto cleanup;
564  }
565  }
566

568  ret =
569  gnutls_certificate_set_x509_key_file(st->x509_cred,

573  if (ret < 0) {
574  ret = -1;
575  rc_log(LOG_ERR,
576  "%s: error in setting X.509 cert and key file: %s",
577  __func__, gnutls_strerror(ret));
578  goto cleanup;
579  }

583  cert_verify_callback);
584  }

587  authservers = rc_conf_srv(rh, "authserver");
588  if (authservers == NULL) {
589  rc_log(LOG_ERR,
590  "%s: cannot find authserver", __func__);
591  ret = -1;
592  goto cleanup;
593  }

595  ret = -1;
596  rc_log(LOG_ERR,
597  "%s: too many auth servers for TLS/DTLS; only one is allowed",
598  __func__);

616  __func__);
617  goto cleanup;

622  ret = -1;
623  rc_log(LOG_ERR,
624  "%s: PSK key is not in 'username@hexkey' format",
625  __func__);
626  goto cleanup;
627  }
628

630  if (username_len + 1 > sizeof(username)) {
631  rc_log(LOG_ERR,
632  "%s: PSK username too big", __func__);
633  ret = -1;
634  goto cleanup;
635  }

638

641  hexkey.size = strlen(p);
642 
643  ret = gnutls_psk_allocate_client_credentials(&st->psk_cred);
644  if (ret < 0) {
645  ret = -1;
646  rc_log(LOG_ERR,
647  "%s: error in setting PSK credentials: %s",
648  __func__, gnutls_strerror(ret));
649  goto cleanup;

653  gnutls_psk_set_client_credentials(st->psk_cred,
654  username, &hexkey,

656  if (ret < 0) {
657  ret = -1;
658  rc_log(LOG_ERR,
659  "%s: error in setting PSK key: %s",
660  __func__, gnutls_strerror(ret));
661  goto cleanup;
662  }

666  if (ret < 0) {

668  goto cleanup;

672  rh->so.sendto = tls_sendto;
673  rh->so.recvfrom = tls_recvfrom;
674  rh->so.lock = tls_lock;

680  deinit_session(&st->ctx);
681  if (st->x509_cred)
682  gnutls_certificate_free_credentials(st->x509_cred);
683  if (st->psk_cred)
684  gnutls_psk_free_client_credentials(st->psk_cred);
685  }
686  free(st);
687  return ret;
688 }
