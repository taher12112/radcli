93 int rc_avpair_assign (VALUE_PAIR *vp, void const *pval, int len)
94 {
95 
96  switch (vp->type)
97  {
98  case PW_TYPE_STRING:
99  if (len == -1)
100  len = (uint32_t)strlen((char const *)pval);
101  if (len > AUTH_STRING_LEN) {
102  rc_log(LOG_ERR, "rc_avpair_assign: bad attribute length");
103  return -1;
104  }
105  memcpy(vp->strvalue, (char const *)pval, len);
106  vp->strvalue[len] = '\0';
107  vp->lvalue = len;
108  break;
109 
110  case PW_TYPE_DATE:
111  case PW_TYPE_INTEGER:
112  case PW_TYPE_IPADDR:
113  vp->lvalue = * (uint32_t *) pval;
114  break;
115  case PW_TYPE_IPV6ADDR:
116  if (len != 16) {
117  rc_log(LOG_ERR, "rc_avpair_assign: bad IPv6 length");
118  return -1;
119  }
120  memcpy(vp->strvalue, (char const *)pval, len);
121  vp->lvalue = len;
122  break;
123 
124  case PW_TYPE_IPV6PREFIX:
125  if (len < 2 || len > 18) {
126  rc_log(LOG_ERR, "rc_avpair_assign: bad IPv6 prefix length");
127  return -1;
128  }
129  memcpy(vp->strvalue, (char const *)pval, len);
130  vp->lvalue = len;
131  break;
132 
133  default:
134  rc_log(LOG_ERR, "rc_avpair_assign: no attribute %d in dictionary", vp->type);
135  return -1;
136  }
137  return 0;
138 }

151 VALUE_PAIR *rc_avpair_new (rc_handle const *rh, int attrid, void const *pval, int len, int vendorpec)
152 {
153  VALUE_PAIR *vp = NULL;
154  DICT_ATTR *pda;
155 
156  attrid = attrid | (vendorpec << 16);
157  if ((pda = rc_dict_getattr (rh, attrid)) == NULL)
158  {
159  rc_log(LOG_ERR,"rc_avpair_new: no attribute %d in dictionary", attrid);
160  return NULL;
161  }
162  if (vendorpec != 0 && rc_dict_getvend(rh, vendorpec) == NULL)
163  {
164  rc_log(LOG_ERR,"rc_avpair_new: no Vendor-Id %d in dictionary", vendorpec);
165  return NULL;
166  }
167  if ((vp = malloc (sizeof (VALUE_PAIR))) != NULL)
168  {
169  strlcpy (vp->name, pda->name, sizeof (vp->name));
170  vp->attribute = attrid;
171  vp->next = NULL;
172  vp->type = pda->type;
173  if (rc_avpair_assign (vp, pval, len) == 0)
174  {
175  /* XXX: Fix up Digest-Attributes */
176  switch (vp->attribute) {
177  case PW_DIGEST_REALM:
178  case PW_DIGEST_NONCE:
179  case PW_DIGEST_METHOD:
180  case PW_DIGEST_URI:
181  case PW_DIGEST_QOP:
182  case PW_DIGEST_ALGORITHM:
183  case PW_DIGEST_BODY_DIGEST:
184  case PW_DIGEST_CNONCE:
185  case PW_DIGEST_NONCE_COUNT:
186  case PW_DIGEST_USER_NAME:
187  /* overlapping! */
188  if (vp->lvalue > AUTH_STRING_LEN - 2)
189  vp->lvalue = AUTH_STRING_LEN - 2;
190  memmove(&vp->strvalue[2], &vp->strvalue[0], vp->lvalue);
191  vp->strvalue[0] = vp->attribute - PW_DIGEST_REALM + 1;
192  vp->lvalue += 2;
193  vp->strvalue[1] = vp->lvalue;
194  vp->strvalue[vp->lvalue] = '\0';
195  vp->attribute = PW_DIGEST_ATTRIBUTES;
196  default:
197  break;
198  }
199  return vp;
200  }
201  free (vp);
202  vp = NULL;
203  }
204  else
205  {
206  rc_log(LOG_CRIT,"rc_avpair_new: out of memory");
207  }
208 
209  return vp;
210 }

223 VALUE_PAIR *rc_avpair_gen(rc_handle const *rh, VALUE_PAIR *pair, unsigned char const *ptr,
224  int length, int vendorpec)
225 {
226  int attribute, attrlen, x_len;
227  unsigned char const *x_ptr;
228  uint32_t lvalue;
229  DICT_ATTR *attr;
230  VALUE_PAIR *rpair;
231  char buffer[(AUTH_STRING_LEN * 2) + 1];
232  /* For hex string conversion. */
233  char hex[3];
234 
235  if (length < 2) {
236  rc_log(LOG_ERR, "rc_avpair_gen: received attribute with "
237  "invalid length");
238  goto shithappens;
239  }
240  attrlen = ptr[1];
241  if (length < attrlen || attrlen < 2) {
242  rc_log(LOG_ERR, "rc_avpair_gen: received attribute with "
243  "invalid length");
244  goto shithappens;
245  }
246 
247  /* Advance to the next attribute and process recursively */
248  if (length != attrlen) {
249  pair = rc_avpair_gen(rh, pair, ptr + attrlen, length - attrlen,
250  vendorpec);
251  if (pair == NULL)
252  return NULL;
253  }
254 
255  /* Actual processing */
256  attribute = ptr[0] | (vendorpec << 16);
257  ptr += 2;
258  attrlen -= 2;
259 
260  /* VSA */
261  if (attribute == PW_VENDOR_SPECIFIC) {
262  if (attrlen < 4) {
263  rc_log(LOG_ERR, "rc_avpair_gen: received VSA "
264  "attribute with invalid length");
265  goto skipit;
266  }
267  memcpy(&lvalue, ptr, 4);
268  vendorpec = ntohl(lvalue);
269  if (rc_dict_getvend(rh, vendorpec) == NULL) {
270  /* Warn and skip over the unknown VSA */
271  rc_log(LOG_WARNING, "rc_avpair_gen: received VSA "
272  "attribute with unknown Vendor-Id %d", vendorpec);
273  goto skipit;
274  }
275  /* Process recursively */
276  return rc_avpair_gen(rh, pair, ptr + 4, attrlen - 4,
277  vendorpec);
278  }
279 
280  /* Normal */
281  attr = rc_dict_getattr(rh, attribute);
282  if (attr == NULL) {
283  buffer[0] = '\0'; /* Initial length. */
284  x_ptr = ptr;
285  for (x_len = attrlen; x_len > 0; x_len--, x_ptr++) {
286  snprintf(hex, sizeof(hex), "%2.2X", x_ptr[0]);
287  strcat(buffer, hex);
288  }
289  if (vendorpec == 0) {
290  rc_log(LOG_WARNING, "rc_avpair_gen: received "
291  "unknown attribute %d of length %d: 0x%s",
292  attribute, attrlen + 2, buffer);
293  } else {
294  rc_log(LOG_WARNING, "rc_avpair_gen: received "
295  "unknown VSA attribute %d, vendor %d of "
296  "length %d: 0x%s", attribute & 0xffff,
297  VENDOR(attribute), attrlen + 2, buffer);
298  }
299  goto skipit;
300  }
301 
302  rpair = calloc(1, sizeof(*rpair));
303  if (rpair == NULL) {
304  rc_log(LOG_CRIT, "rc_avpair_gen: out of memory");
305  goto shithappens;
306  }
307 
308  /* Insert this new pair at the beginning of the list */
309  rpair->next = pair;
310  pair = rpair;
311  strcpy(pair->name, attr->name);
312  pair->attribute = attr->value;
313  pair->type = attr->type;
314 
315  switch (attr->type) {
316  case PW_TYPE_STRING:
317  memcpy(pair->strvalue, (char *)ptr, (size_t)attrlen);
318  pair->strvalue[attrlen] = '\0';
319  pair->lvalue = attrlen;
320  break;
321 
322  case PW_TYPE_INTEGER:
323  if (attrlen != 4) {
324  rc_log(LOG_ERR, "rc_avpair_gen: received INT "
325  "attribute with invalid length");
326  goto skipit;
327  }
328  case PW_TYPE_IPADDR:
329  if (attrlen != 4) {
330  rc_log(LOG_ERR, "rc_avpair_gen: received IPADDR"
331  " attribute with invalid length");
332  goto skipit;
333  }
334  memcpy((char *)&lvalue, (char *)ptr, 4);
335  pair->lvalue = ntohl(lvalue);
336  break;
337  case PW_TYPE_IPV6ADDR:
338  if (attrlen != 16) {
339  rc_log(LOG_ERR, "rc_avpair_gen: received IPV6ADDR"
340  " attribute with invalid length");
341  goto skipit;
342  }
343  memcpy(pair->strvalue, (char *)ptr, 16);
344  pair->lvalue = attrlen;
345  break;
346  case PW_TYPE_IPV6PREFIX:
347  if (attrlen > 18 || attrlen < 2) {
348  rc_log(LOG_ERR, "rc_avpair_gen: received IPV6PREFIX"
349  " attribute with invalid length: %d", attrlen);
350  goto skipit;
351  }
352  memcpy(pair->strvalue, (char *)ptr, attrlen);
353  pair->lvalue = attrlen;
354  break;
355  case PW_TYPE_DATE:
356  if (attrlen != 4) {
357  rc_log(LOG_ERR, "rc_avpair_gen: received DATE "
358  "attribute with invalid length");
359  goto skipit;
360  }
361 
362  default:
363  rc_log(LOG_WARNING, "rc_avpair_gen: %s has unknown type",
364  attr->name);
365  goto skipit;
366  }
367 
368 skipit:
369  return pair;
370 
371 shithappens:
372  while (pair != NULL) {
373  rpair = pair->next;
374  free(pair);
375  pair = rpair;
376  }
377  return NULL;
378 }

406 void rc_avpair_insert(VALUE_PAIR **a, VALUE_PAIR *p, VALUE_PAIR *b)
407 {
408  VALUE_PAIR *this_node = NULL;
409  VALUE_PAIR *vp;
410 
411  if (b->next != NULL)
412  {
413  rc_log(LOG_CRIT, "rc_avpair_insert: value pair (0x%p) next ptr. (0x%p) not NULL", b, b->next);
414  abort ();
415  }
416 
417  if (*a == NULL)
418  {
419  *a = b;
420  return;
421  }
422 
423  vp = *a;
424 
425  if ( p == NULL) /* run to end of "a" list */
426  {
427  while (vp != NULL)
428  {
429  this_node = vp;
430  vp = vp->next;
431  }
432  }
433  else /* look for the "p" entry in the "a" list */
434  {
435  this_node = *a;
436  while (this_node != NULL)
437  {
438  if (this_node == p)
439  {
440  break;
441  }
442  this_node = this_node->next;
443  }
444  }
445 
446  b->next = this_node->next;
447  this_node->next = b;
448 
449  return;
450 }

478 static void rc_fieldcpy(char *string, char const **uptr, char const *stopat, size_t len)
479 {
480  char const *ptr, *estring;
481 
482  ptr = *uptr;
483  estring = string + len - 1;
484  if (*ptr == '"')
485  {
486  ptr++;
487  while (*ptr != '"' && *ptr != '\0' && *ptr != '\n')
488  {
489  if (string < estring)
490  *string++ = *ptr;
491  ptr++;
492  }
493  if (*ptr == '"')
494  {
495  ptr++;
496  }
497  *string = '\0';
498  *uptr = ptr;
499  return;
500  }
501 
502  while (*ptr != '\0' && strchr(stopat, *ptr) == NULL)
503  {
504  if (string < estring)
505  *string++ = *ptr;
506  ptr++;
507  }
508  *string = '\0';
509  *uptr = ptr;
510  return;
511 }

520 int rc_avpair_parse (rc_handle const *rh, char const *buffer, VALUE_PAIR **first_pair)
521 {
522  int mode;
523  char attrstr[AUTH_ID_LEN];
524  char valstr[AUTH_STRING_LEN + 1], *p;
525  DICT_ATTR *attr = NULL;
526  DICT_VALUE *dval;
527  VALUE_PAIR *pair;
528  VALUE_PAIR *link;
529  struct tm *tm;
530  time_t timeval;
531 
532  mode = PARSE_MODE_NAME;
533  while (*buffer != '\n' && *buffer != '\0')
534  {
535  if (*buffer == ' ' || *buffer == '\t')
536  {
537  buffer++;
538  continue;
539  }
540 
541  switch (mode)
542  {
543  case PARSE_MODE_NAME: /* Attribute Name */
544  rc_fieldcpy (attrstr, &buffer, " \t\n=,", sizeof(attrstr));
545  if ((attr =
546  rc_dict_findattr (rh, attrstr)) == NULL)
547  {
548  rc_log(LOG_ERR, "rc_avpair_parse: unknown attribute");
549  if (*first_pair) {
550  rc_avpair_free(*first_pair);
551  *first_pair = NULL;
552  }
553  return -1;
554  }
555  mode = PARSE_MODE_EQUAL;
556  break;
557 
558  case PARSE_MODE_EQUAL: /* Equal sign */
559  if (*buffer == '=')
560  {
561  mode = PARSE_MODE_VALUE;
562  buffer++;
563  }
564  else
565  {
566  rc_log(LOG_ERR, "rc_avpair_parse: missing or misplaced equal sign");
567  if (*first_pair) {
568  rc_avpair_free(*first_pair);
569  *first_pair = NULL;
570  }
571  return -1;
572  }
573  break;
574 
575  case PARSE_MODE_VALUE: /* Value */
576  rc_fieldcpy (valstr, &buffer, " \t\n,", sizeof(valstr));
577 
578  if ((pair = malloc (sizeof (VALUE_PAIR))) == NULL)
579  {
580  rc_log(LOG_CRIT, "rc_avpair_parse: out of memory");
581  if (*first_pair) {
582  rc_avpair_free(*first_pair);
583  *first_pair = NULL;
584  }
585  return -1;
586  }
587  strcpy (pair->name, attr->name);
588  pair->attribute = attr->value;
589  pair->type = attr->type;
590 
591  switch (pair->type)
592  {
593 
594  case PW_TYPE_STRING:
595  strcpy (pair->strvalue, valstr);
596  pair->lvalue = (uint32_t)strlen(valstr);
597  break;
598 
599  case PW_TYPE_INTEGER:
600  if (isdigit (*valstr))
601  {
602  pair->lvalue = atoi (valstr);
603  }
604  else
605  {
606  if ((dval = rc_dict_findval (rh, valstr))
607  == NULL)
608  {
609  rc_log(LOG_ERR, "rc_avpair_parse: unknown attribute value: %s", valstr);
610  if (*first_pair) {
611  rc_avpair_free(*first_pair);
612  *first_pair = NULL;
613  }
614  free (pair);
615  return -1;
616  }
617  else
618  {
619  pair->lvalue = dval->value;
620  }
621  }
622  break;
623 
624  case PW_TYPE_IPADDR:
625  if (inet_pton(AF_INET, valstr, &pair->lvalue) == 0) {
626  rc_log(LOG_ERR, "rc_avpair_parse: invalid IPv4 address %s", valstr);
627  free(pair);
628  return -1;
629  }
630 
631  pair->lvalue = ntohl(pair->lvalue);
632  break;
633 
634  case PW_TYPE_IPV6ADDR:
635  if (inet_pton(AF_INET6, valstr, pair->strvalue) == 0) {
636  rc_log(LOG_ERR, "rc_avpair_parse: invalid IPv6 address %s", valstr);
637  free(pair);
638  return -1;
639  }
640  pair->lvalue = 16;
641  break;
642 
643  case PW_TYPE_IPV6PREFIX:
644  p = strchr(valstr, '/');
645  if (p == NULL) {
646  rc_log(LOG_ERR, "rc_avpair_parse: invalid IPv6 prefix %s", valstr);
647  free(pair);
648  return -1;
649  }
650  *p = 0;
651  p++;
652  pair->strvalue[0] = 0;
653  pair->strvalue[1] = atoi(p);
654 
655  if (inet_pton(AF_INET6, valstr, pair->strvalue+2) == 0) {
656  rc_log(LOG_ERR, "rc_avpair_parse: invalid IPv6 prefix %s", valstr);
657  free(pair);
658  return -1;
659  }
660  pair->lvalue = 2+16;
661  break;
662 
663  case PW_TYPE_DATE:
664  timeval = time (0);
665  tm = localtime (&timeval);
666  tm->tm_hour = 0;
667  tm->tm_min = 0;
668  tm->tm_sec = 0;
669  rc_str2tm (valstr, tm);
670 #ifdef TIMELOCAL
671  pair->lvalue = (uint32_t) timelocal (tm);
672 #else /* TIMELOCAL */
673  pair->lvalue = (uint32_t) mktime (tm);
674 #endif /* TIMELOCAL */
675  break;
676 
677  default:
678  rc_log(LOG_ERR, "rc_avpair_parse: unknown attribute type %d", pair->type);
679  if (*first_pair) {
680  rc_avpair_free(*first_pair);
681  *first_pair = NULL;
682  }
683  free (pair);
684  return -1;
685  }
686 
687  /* XXX: Fix up Digest-Attributes */
688  switch (pair->attribute) {
689  case PW_DIGEST_REALM:
690  case PW_DIGEST_NONCE:
691  case PW_DIGEST_METHOD:
692  case PW_DIGEST_URI:
693  case PW_DIGEST_QOP:
694  case PW_DIGEST_ALGORITHM:
695  case PW_DIGEST_BODY_DIGEST:
696  case PW_DIGEST_CNONCE:
697  case PW_DIGEST_NONCE_COUNT:
698  case PW_DIGEST_USER_NAME:
699  /* overlapping! */
700  if (pair->lvalue > AUTH_STRING_LEN - 2)
701  pair->lvalue = AUTH_STRING_LEN - 2;
702  memmove(&pair->strvalue[2], &pair->strvalue[0], pair->lvalue);
703  pair->strvalue[0] = pair->attribute - PW_DIGEST_REALM + 1;
704  pair->lvalue += 2;
705  pair->strvalue[1] = pair->lvalue;
706  pair->strvalue[pair->lvalue] = '\0';
707  pair->attribute = PW_DIGEST_ATTRIBUTES;
708  break;
709  default:
710  break;
711  }
712 
713  pair->next = NULL;
714 
715  if (*first_pair == NULL)
716  {
717  *first_pair = pair;
718  }
719  else
720  {
721  link = *first_pair;
722  while (link->next != NULL)
723  {
724  link = link->next;
725  }
726  link->next = pair;
727  }
728 
729  mode = PARSE_MODE_NAME;
730  break;
731 
732  default:
733  mode = PARSE_MODE_NAME;
734  break;
735  }
736  }
737  return 0;
738 }

750 int rc_avpair_tostr (rc_handle const *rh, VALUE_PAIR *pair, char *name, int ln, char *value, int lv)
751 {
752  DICT_VALUE *dval;
753  struct in_addr inad;
754  unsigned char *ptr;
755  unsigned int pos;
756 
757  *name = *value = '\0';
758 
759  if (!pair || pair->name[0] == '\0') {
760  rc_log(LOG_ERR, "rc_avpair_tostr: pair is NULL or empty");
761  return -1;
762  }
763 
764  strlcpy(name, pair->name, (size_t) ln);
765 
766  switch (pair->type)
767  {
768  case PW_TYPE_STRING:
769  lv--;
770  pos = 0;
771  ptr = (unsigned char *) pair->strvalue;
772  if (pair->attribute == PW_DIGEST_ATTRIBUTES) {
773  pair->strvalue[*(ptr + 1)] = '\0';
774  ptr += 2;
775  }
776  while (*ptr != '\0')
777  {
778  if (!(isprint (*ptr)))
779  {
780  if (lv >= 4) {
781  snprintf (&value[pos], lv, "\\%03o", *ptr);
782  pos += 4;
783  lv -= 4;
784  } else {
785  break;
786  }
787  }
788  else
789  {
790  if (lv > 0) {
791  value[pos++] = *ptr;
792  lv--;
793  } else {
794  break;
795  }
796  }
797  ptr++;
798  }
799  if (lv > 0)
800  value[pos++] = 0;
801  else
802  value[pos-1] = 0;
803  break;
804 
805  case PW_TYPE_INTEGER:
806  dval = rc_dict_getval (rh, pair->lvalue, pair->name);
807  if (dval != NULL)
808  {
809  strlcpy(value, dval->name, (size_t) lv);
810  }
811  else
812  {
813  snprintf(value, lv, "%ld", (long int)pair->lvalue);
814  }
815  break;
816 
817  case PW_TYPE_IPADDR:
818  inad.s_addr = htonl(pair->lvalue);
819  strlcpy (value, inet_ntoa (inad), (size_t) lv);
820  break;
821 
822  case PW_TYPE_IPV6ADDR:
823  if (inet_ntop(AF_INET6, pair->strvalue, value, lv) == NULL)
824  return -1;
825  break;
826 
827  case PW_TYPE_IPV6PREFIX: {
828  uint8_t ip[16];
829  uint8_t txt[48];
830  if (pair->lvalue < 2)
831  return -1;
832 
833  memset(ip, 0, sizeof(ip));
834  memcpy(ip, pair->strvalue+2, pair->lvalue-2);
835 
836  if (inet_ntop(AF_INET6, ip, (void*)txt, sizeof(txt)) == NULL)
837  return -1;
838  snprintf(value, lv, "%s/%u", txt, (unsigned)pair->strvalue[1]);
839 
840  break;
841  }
842  case PW_TYPE_DATE:
843  strftime (value, lv, "%m/%d/%y %H:%M:%S",
844  gmtime ((time_t *) & pair->lvalue));
845  break;
846 
847  default:
848  rc_log(LOG_ERR, "rc_avpair_tostr: unknown attribute type %d", pair->type);
849  return -1;
850  break;
851  }
852
