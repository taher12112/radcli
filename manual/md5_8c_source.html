102  while (len >= MD5_BLOCK_LENGTH) {

116  * 1 0* (64-bit count of bits processed, MSB-first)

131  if (padlen < 1 + 8)

133  MD5Update(ctx, PADDING, padlen - 8); /* padlen - 8 <= 64 */
134  MD5Update(ctx, count, 8);

136  if (digest != NULL) {

139  }
