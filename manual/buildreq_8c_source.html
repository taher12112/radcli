108 int rc_aaa_ctx_server(rc_handle *rh, RC_AAA_CTX **ctx, SERVER *aaaserver,
109  rc_type type,
110  uint32_t client_port,
111  VALUE_PAIR *send, VALUE_PAIR **received,
112  char *msg, int add_nas_port, rc_standard_codes request_type)
113 {
114  SEND_DATA data;
115  VALUE_PAIR *adt_vp = NULL;
116  int result;
117  int timeout = rc_conf_int(rh, "radius_timeout");
118  int retries = rc_conf_int(rh, "radius_retries");
119  double start_time = 0;
120  double now = 0;
121  time_t dtime;
122  int servernum;
123 
124  data.send_pairs = send;
125  data.receive_pairs = NULL;
126 
127  /*
128  * if there is more than zero servers, then divide waiting time
129  * among all the servers.
130  */
131  if(aaaserver->max > 0) {
132  if(timeout > 0) {
133  timeout = (timeout+1) / aaaserver->max;
134  }
135  if(retries > 0) {
136  retries = (retries+1) / aaaserver->max;
137  }
138  }
139 
140  if (add_nas_port != 0 && rc_avpair_get(data.send_pairs, PW_NAS_PORT, 0) == NULL) {
141  /*
142  * Fill in NAS-Port
143  */
144  if (rc_avpair_add(rh, &(data.send_pairs), PW_NAS_PORT,
145  &client_port, 0, 0) == NULL)
146  return ERROR_RC;
147  }
148 
149  if (request_type == PW_ACCOUNTING_REQUEST) {
150  /*
151  * Fill in Acct-Delay-Time
152  */
153  dtime = 0;
154  now = rc_getctime();
155  adt_vp = rc_avpair_get(data.send_pairs, PW_ACCT_DELAY_TIME, 0);
156  if (adt_vp == NULL) {
157  adt_vp = rc_avpair_add(rh, &(data.send_pairs),
158  PW_ACCT_DELAY_TIME, &dtime, 0, 0);
159  if (adt_vp == NULL)
160  return ERROR_RC;
161  start_time = now;
162  } else {
163  start_time = now - adt_vp->lvalue;
164  }
165  }
166 
167  if (data.receive_pairs != NULL) {
168  rc_avpair_free(data.receive_pairs);
169  data.receive_pairs = NULL;
170  }
171 
172  servernum=0;
173  do {
174  rc_buildreq(rh, &data, request_type, aaaserver->name[servernum],
175  aaaserver->port[servernum],
176  aaaserver->secret[servernum], timeout, retries);
177 
178  if (request_type == PW_ACCOUNTING_REQUEST) {
179  dtime = rc_getctime() - start_time;
180  rc_avpair_assign(adt_vp, &dtime, 0);
181  }
182 
183  result = rc_send_server_ctx (rh, ctx, &data, msg, type);
184 
185  if (request_type != PW_ACCOUNTING_REQUEST) {
186  *received = data.receive_pairs;
187  } else {
188  rc_avpair_free(data.receive_pairs);
189  }
190 
191  if(result == OK_RC) {
192  DEBUG(LOG_INFO,
193  "servernum %u returned success", servernum);
194  return result;
195  }
196 
197  //rc_log(LOG_ERR,
198  // "servernum %u returned error: %d", servernum, result);
199  servernum++;
200  } while(servernum < aaaserver->max && result == TIMEOUT_RC);
201 
202  return result;
203 }

296 int rc_check(rc_handle *rh, char *host, char *secret, unsigned short port, char *msg)
297 {
298  SEND_DATA data;
299  int result;
300  uint32_t service_type;
301  int timeout = rc_conf_int(rh, "radius_timeout");
302  int retries = rc_conf_int(rh, "radius_retries");
303  rc_type type;
304 
305  data.send_pairs = data.receive_pairs = NULL;
306 
307  if (rh->so_type == RC_SOCKET_TLS || rh->so_type == RC_SOCKET_DTLS)
308  type = AUTH;
309  else
310  type = ACCT;
311 
312  /*
313  * Fill in Service-Type
314  */
315 
316  service_type = PW_ADMINISTRATIVE;
317  rc_avpair_add(rh, &(data.send_pairs), PW_SERVICE_TYPE, &service_type, 0, 0);
318 
319  rc_buildreq(rh, &data, PW_STATUS_SERVER, host, port, secret, timeout, retries);
320  result = rc_send_server (rh, &data, msg, type);
321 
322  rc_avpair_free(data.receive_pairs);
323 
324  return result;
325 }
